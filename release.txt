Releasing a new version

# make sure things are not broken and that files match up
  make rcheck
  
# update the version in grheat/__init__.py
  git commit grheat/__init__.py -m 'update version'

# update CHANGELOG.txt  `git shortlog 0.5.0..HEAD`
  git commit CHANGELOG.txt -m 'update recent changes' 
  git push

# update lite implementation
  make lite
  make lite-deploy

# create release tag on github
  git tag 0.5.0
  git push origin 0.5.0

# manually publish release at
  open https://github.com/scottprahl/grheat

# verify action success
  open https://pypi.org/project/grheat/
  open https://grheat.readthedocs.io

# wait for feedstock update
  open https://github.com/conda-forge/grheat-feedstock
